1 // Vaatii cbNetwork ja node-optimist-paketin: https://github.com/substack/node-optimist
  2 /**#nocode+*/
  3 var argv = require('optimist')
  4     .default({p : 1337, a : undefined})
  5     .alias({'p' : 'port', 'a' : 'address', 'd' : 'debug'})
  6     .argv
  7   , NET = require('./Constants').NET
  8   , NetMessages = require('./NetMessage')
  9   , NetMatch = require('./NetMatch');
 10 /**#nocode-*/
 11 
 12 // Tehdään uusi palvelin.
 13 /** @ignore */
 14 var server = new NetMatch({ port: argv.p, address: argv.a});
 15 
 16 // Tehdään jotain sitä mukaa, kun dataa tulloo sissään.
 17 server.on('message', function (client) {
 18   var data = client.data
 19   var msgType = data.getByte();
 20   
 21   // Jos oli uusi pelaaja
 22   if (msgType === NET.LOGIN) {
 23     server.login(client);
 24     return;
 25   }
 26   
 27   // Luetaan lähetetty pelaajan ID, joka on pelaajan järjestysnumero ja aina väliltä 1...MAX_PLAYERS
 28   var currentPlayerId = data.getByte();
 29   
 30   // Haetaan pelaajan instanssi Player-luokasta
 31   var player = this.players[currentPlayerId];
 32   
 33   console.log(player);
 34   return;
 35   
 36   // Luetaan dataa
 37   var breakOut = false;
 38   while (msgType && !breakOut) {
 39     switch (msgType) {
 40       case NET.PLAYER:
 41         break;
 42       case NET.PLAYERNAME:
 43         break;
 44       case NET.TEXTMESSAGE:
 45         break;
 46       case NET.MAPCHANGE:
 47         break;
 48       default:
 49         breakOut = true;
 50         continue; // eli oikeasti break while-loopista
 51     }
 52     msgType = data.getByte();
 53   }
 54 });
